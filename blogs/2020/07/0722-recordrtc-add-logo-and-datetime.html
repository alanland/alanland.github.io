    RecordRTC添加当前时间戳和Logo

    Alan 7/24/2020 webrtc

    RecordRtc 录像保存的视频需要添加录像时间等信息。

    前端方案就是录制的时候直接写进视频对象（Canvas），后端方案是视频保存的时候保存相关信息进数据库，再用 FFMPG 等工具进行视频处理。

<script>
var videoPreview = document.getElementById('video-preview');
videoPreview.controls = false
var logoImage = document.getElementById('logo-image');
var waitImage = document.getElementById('wait-image');
var mediaBox = document.getElementById('media-box')
var mediaStatus = document.getElementById('media-status')
var mediaImg = document.getElementById('media-img')
var formChecked = document.getElementById('formChecked')
var formInput = document.getElementById('formInput')

var watermarkTimestamp = formChecked.checked
var watermarkText = ''

navigator.mediaDevices.getUserMedia({video: true, audio: true}).then(function (camera) {
  var canvas = document.createElement('canvas');
  var context = canvas.getContext('2d');

  canvas.style = 'position: absolute; top: 0; left: 0; opacity: 0; margin-top: -9999999999; margin-left: -9999999999; top: -9999999999; left: -9999999999; z-index: -1;';
  document.body.appendChild(canvas);

  var video = document.createElement('video');
  video.autoplay = true;
  video.playsinline = true;
  video.srcObject = camera;

  var canvasStream = canvas.captureStream(15);

  var audioPlusCanvasStream = new MediaStream();

  // "getTracks" is RecordRTC's built-in function
  getTracks(canvasStream, 'video').forEach(function (videoTrack) {
    audioPlusCanvasStream.addTrack(videoTrack);
  });

  // "getTracks" is RecordRTC's built-in function
  getTracks(camera, 'audio').forEach(function (audioTrack) {
    audioPlusCanvasStream.addTrack(audioTrack);
  });

  var recorder = RecordRTC(audioPlusCanvasStream, {
    type: 'video',
    timeSlice: 1000, // pass this parameter
    onTimeStamp: function (timestamp, timestamps) {
      var duration = (new Date().getTime() - timestamps[0]) / 1000;
      if (duration < 0) return;
      document.querySelector('#timestamp').innerHTML = '' + duration;
    }
  });

  window.recorder = recorder

  // recorder.startRecording();

  videoPreview.srcObject = canvasStream;


  var tries = 0;
  (function looper() {
    if (!recorder) return; // ignore/skip on stop-recording

    tries += 100;

    canvas.width = 320;
    canvas.height = 240;

    // show hello.png for one second
    context.drawImage(video, 0, 0, canvas.width, canvas.height);
    context.drawImage(logoImage, 4, 4, 20, 20)

    height = 16
    paddingX = 34
    paddingY = 2

    function rectY(line) {
      return paddingY + height * (line - 1) // + (line == 1 ? 2 : 0)
    }

    function rectH(line) {
      return 16 // + (line == 1 ? 2 : 0)
    }

    function textY(line) {
      return line * height - 1
    }

    function draw(line, text) {
      if (recorder && (recorder.state === 'recording' || recorder.state === 'paused')) {
        context.font = "normal 14px Verdana";
        context.fillStyle = "#00000066";
        context.fillRect(paddingX, rectY(line), context.measureText(text).width + 2, rectH(line));
        context.fillStyle = "#FFFFFF";
        context.fillText(text, paddingX, textY(line));
      }
    }

    if (watermarkTimestamp) {
      draw(1, new Date().toISOString().replace('T', ' ').replace('Z', ''))
    }
    if (watermarkText) {
      draw(2, watermarkText)
      draw(3, watermarkText + 'xxxxxxsntho')
      draw(4, watermarkText)
    }

    // repeat (looper)
    setTimeout(looper, 100);
  })();

  document.getElementById('btn-start-recording').onclick = function () {
    watermarkTimestamp = formChecked.checked
    if (recorder.state === 'recording' || recorder.state === 'paused') {
      recorder.stopRecording(function (u) {
        recorder.startRecording()
      })
    } else {
      recorder.startRecording()
    }

    videoPreview.srcObject = canvasStream
    mediaBox.className = 'recording'
    imgRecord()
  }
  document.getElementById('btn-stop-recording').onclick = function () {
    recorder.stopRecording(function (uuu) {
      filename = `${new Date().toISOString().substr(0, 20)}.webm` + '.webm'
      console.log(filename)
      recorder.save(filename)
      var blob = recorder.getBlob();
      // recorder = null;
      // camera.stop();

      watermarkText = ''
      watermarkTimestamp = false
      videoPreview.srcObject = canvasStream
      imgStop()
      // videoPreview.srcObject = null;
      // videoPreview.src = URL.createObjectURL(blob);
    })
  }
  document.getElementById('btn-pause-recording').onclick = function () {
    if (recorder.getState() === "recording") {
      recorder.pauseRecording()
      videoPreview.pause()
      imgPause()
    }
  }
  document.getElementById('btn-resume-recording').onclick = function () {
    if (recorder.getState() === "paused") {
      recorder.resumeRecording()
      videoPreview.play()
      imgRecord()
    }
  }
  document.getElementById('btn-enable-recording').onclick = function () {
    if (recorder === null) {
      recorder = RecordRTC(audioPlusCanvasStream, {
        type: 'video',
        timeSlice: 1000, // pass this parameter
        onTimeStamp: function (timestamp, timestamps) {
          var duration = (new Date().getTime() - timestamps[0]) / 1000;
          if (duration < 0) return;
          document.querySelector('#timestamp').innerHTML = '' + duration;
        }
      });
      videoPreview.srcObject = null
      mediaImg.src = null
    }
  }
  document.getElementById('btn-close-recording').onclick = function () {
    if (recorder.state === 'recording') {
      recorder.stopRecording(function () {
        recorder = null
        videoPreview.srcObject = null
        mediaImg.src = null
        camera.stop()
      })
    } else {
      recorder = null
      videoPreview.srcObject = null
      mediaImg.src = null
      camera.stop()
    }
  }
  formChecked.onchange = function (e) {
    watermarkTimestamp = e.target.checked
  }
  document.getElementById('formButton').onclick = function () {
    watermarkText = formInput.value
  }

  function imgStop() {
    mediaImg.src = './images/stoped.gif'
  }

  function imgRecord() {
    mediaImg.src = './images/progress.gif'
  }

  function imgPause() {
    mediaImg.src = './images/pause.gif'
  }


}).catch(function (error) {
  alert('Unable to capture camera. Please check console logs.');
  console.error(error);
});
